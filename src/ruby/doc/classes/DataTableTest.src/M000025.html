  def test_read
    f1 = File.new("tmp/#{@data_table.docid}", "w+")
    @data_table.read do |buffer|
      f1.write buffer
    end
    f1.close
    assert_equal  File.size("test/fixtures/#{@data_table.id}"),
                    File.size("tmp/#{@data_table.id}")
  end
