  def squery(squery)
    response = metacat_get({
      'action'  =>  'squery',
      'qformat' =>  'xml',
      'query'   =>  squery
    })
    if(response.content_type == 'text/xml')
      response.read_body
    else
      raise "Metacat returned unexpected Content Type"
    end
  end
